\chapter*{Liste des symboles}
\addcontentsline{toc}{chapter}{Liste des symboles}

\begin{tabular}{@{}lp{14cm}@{}}
$s(x)$               & Score d'anomalie d'un point $x$ dans Isolation Forest \\[3pt]
$h(x)$               & Profondeur (nombre de coupures) nécessaire pour isoler $x$
                        dans un arbre \\[3pt]
$E[h(x)]$            & Espérance de la profondeur d'isolation sur l'ensemble des
                        arbres de la forêt \\[3pt]
$c(n)$               & Facteur de normalisation : profondeur moyenne d'isolation
                        dans un arbre binaire de $n$ observations \\[3pt]
$n$                   & Nombre d'observations (bins temporels) \\[3pt]
$K$                   & Nombre de bins dans la fenêtre de persistance :
                        $\lceil \texttt{persist\_hours} \times 60 /
                        \texttt{interval\_min} \rceil$ \\[3pt]
$\sigma$              & Écart-type \\[3pt]
$\tilde{\mu}$         & Médiane d'un signal \\[3pt]
$z_i$                 & Score z robuste (basé sur la MAD) pour le bin $i$ \\[3pt]
$\text{IoU}$          & \textit{Intersection over Union} :
                        $\frac{|A \cap B|}{|A \cup B|}$, mesure de recouvrement
                        temporel entre un épisode prédit $A$ et une fenêtre
                        d'injection $B$ \\[3pt]
$S$                   & Score de sélection pondéré pour le classement des
                        configurations (éq.~\ref{eq:selection_score}) \\[3pt]
$\overline{\text{IoU20}}$ & Taux moyen de détection avec IoU $\geq$ 0.20 \\[3pt]
$\overline{\text{det}}$   & Taux moyen de détection (recouvrement quelconque) \\[3pt]
$\overline{\text{leak}}$  & Taux moyen de fuite sur les scénarios non détectables \\[3pt]
$\overline{\text{FN}}$    & Taux moyen de fausses notifications par vache-jour \\[3pt]
$\sigma_{\text{FN}}$      & Écart-type des fausses notifications entre seeds \\[3pt]
\end{tabular}
